<template>    
    <div>
                  <b-table small striped hover  head-variant="dark" responsive caption-top :fields="fields"
                   :items="datas.content" current-page=1 per-page=7 >
                <template v-slot:table-caption>
                    list of other sellers (table)
                </template>                 
                <template v-slot:cell(index)="data">
                    {{data.index+1}}
                </template>  
          </b-table>
         
    </div>
</template>
<script>

import i18next from 'i18next'

export default {
    name:'f2fTable',
    props:{
            datas : Object
          } ,
    data(){
      return {
          fields: this.fieldz(),
      }
    } ,
    methods:{} ,
    mounted(){},
    computed:{
          fieldz : function(){
              if(this.datas===null) return [];
              if(this.datas.content===undefined)
                return;
              let objs = this.datas.content[0] ;
              let fldz = [];
              for(var key in objs){
                let b = new Object();
                b.key = key;
                b.label = i18next.t('CONTENT_SEARCH_'+key);
                fldz.push(b);
              }
              return fldz;
          } 
    }        
}
</script>
